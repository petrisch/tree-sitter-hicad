START   59

REM string compare
IF $FOO = "foo" THEN
    REM Test
    %XY:=42
IFEND

REM num compare nested not
IF FOO <= 84 THEN
    REM Test
    %YX:=23
   IF NOT WAHR THEN
      %YX:=23
      REM Test
   IFEND
IFEND

REM bool compare nested char compare
IF WAHR THEN
   WAIT Test
  IF $FOO <> "Test" THEN
     WAIT Test
  ELSE
     WAIT More
  IFEND
IFEND

REM sys compare
IF @23=5 THEN
    WAIT 3
IFEND

REM sys char compare
IF $@A3="test" THEN
    WAIT 3
IFEND

REM combine
IF $X="test" AND %Y>0 THEN
    WAIT 3
IFEND

REM arithmetic
IF 5+5=10 THEN
    WAIT 5
IFEND

REM Num comparison
IF SP=1 THEN
 WAIT 6
IFEND

IF %ABR=1 GOTO 99

REM Menu interrupted by condition
OPTION   1  25
  POINT   R
  IF $BHF="A1" OR $BHF="B1" THEN
    POINT   162
  ELSE
    POINT   162
  IFEND
ESC

END
